export async function sha256Hex(text: string): Promise<string> {
  const data = new TextEncoder().encode(text);
  const buf = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(buf))
    .map(b => b.toString(16).padStart(2, '0')).join('');
}

export function DayNote({ note, onUpdate }: { note: string; onUpdate: (n: string) => void }) {
  const [local, setLocal] = useState(note);
  const timer = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => { setLocal(note); }, [note]); // sync on day change

  function handleChange(e: ChangeEvent<HTMLTextAreaElement>) {
    setLocal(e.target.value);
    if (timer.current) clearTimeout(timer.current);
    timer.current = setTimeout(() => onUpdate(e.target.value), 300);
  }

  return (
    <section className="day-note">
      <textarea className="day-note__textarea" value={local}
        onChange={handleChange} placeholder="Notes for today..." rows={3} />
    </section>

function App() {
  const { session, users, register, login, logout, updateAvatar, updatePassword } = useAuth();

}

// Separate component — needed so useAppData is never called conditionally
function AuthenticatedApp({ session, users, onLogout, onUpdateAvatar, onUpdatePassword }) {
  const { days, settings, ..., updateDayNote } = useAppData(session.userId);
  const currentDate = useCurrentDay(settings);
  const [page, setPage] = useState<Page>('today');
  const [sidebarOpen, setSidebarOpen] = useState(true);

  useEffect(() => { applyTheme(settings.theme); }, [settings.theme]);

  const currentDay = days[currentDate] ?? { date: currentDate, shifts: [], expenses: [] };
  const currentUser = users.find(u => u.id === session.userId) ?? null;

  if (page === 'settings') return <SettingsPage settings={settings} onUpdateSettings={updateSettings} onBack={() => setPage('today')} />;
  if (page === 'profile' && currentUser) return <ProfilePage user={currentUser} onUpdateAvatar={onUpdateAvatar} onUpdatePassword={onUpdatePassword} onBack={() => setPage('today')} />;

  return (
    <div className="app">
      <Sidebar isOpen={sidebarOpen} onToggle={...} currentDay={currentDay} settings={settings} />
      <div className="app-content">
        <Header currentDate={currentDate} session={session} avatar={currentUser?.avatar ?? null} onNavigate={setPage} onLogout={onLogout} />
        <main className="main"><div className="main-inner">
          <DayView day={currentDay} settings={settings}
            onAddShift={...} onUpdateShift={...} onRemoveShift={...}
            onAddExpense={...} onUpdateExpense={...} onRemoveExpense={...}
            onUpdateNote={(note) => updateDayNote(currentDate, note)} />
        </div></main>
      </div>
    </div>
  );
}

export function useGymData(userId: string) {
  const [gymProgram, setGymProgram] = useLocalStorage<GymProgram | null>(
    `todaystracker_gym_program_${userId}`, null
  );
  const [gymSessions, setGymSessions] = useLocalStorage<Record<string, GymSession>>(
    `todaystracker_gym_sessions_${userId}`, {}
  );
  function updateGymProgram(program: GymProgram | null): void { setGymProgram(program); }
  function upsertGymSession(date: string, session: GymSession): void {
    setGymSessions({ ...gymSessions, [date]: session });
  }
  return { gymProgram, gymSessions, updateGymProgram, upsertGymSession };
}

import { useLocalStorage } from './useLocalStorage';
import type { GymProgram, GymSession } from '../types';

export function useGymData(userId: string) {
  const [gymProgram, setGymProgram] = useLocalStorage<GymProgram | null>(
    `todaystracker_gym_program_${userId}`, null
  );
  const [gymSessions, setGymSessions] = useLocalStorage<Record<string, GymSession>>(
    `todaystracker_gym_sessions_${userId}`, {}
  );

  function updateGymProgram(program: GymProgram | null): void {
    setGymProgram(program);
  }

  function upsertGymSession(date: string, session: GymSession): void {
    setGymSessions({ ...gymSessions, [date]: session });
  }

  return { gymProgram, gymSessions, updateGymProgram, upsertGymSession };
}
